25void lcd_send_cmd(char cmd)
26{
27 char data_u, data_l;
28 uint8_t data_t[4];
29 data_u = (cmd & 0xf0);
30 data_l = ((cmd << 4) & 0xf0);
31 data_t[0] = data_u | 0x0C; //en=1, rs=0
32 data_t[1] = data_u | 0x08; //en=0, rs=0
33 data_t[2] = data_l | 0x0C; //en=1, rs=0
34 data_t[3] = data_l | 0x08; //en=0, rs=0
35 HAL_I2C_Master_Transmit(&hi2c1, SLAVE_ADDRESS_LCD, (uint8_t *)data_t, 4, 100);
36}

67void lcd_init(void)
68{
69 // 4 bit initialisation
70 HAL_Delay(50); // wait for >40ms

73 lcd_send_cmd(0x30);
74 HAL_Delay(10); // wait for >100us
75 lcd_send_cmd(0x30);
76 HAL_Delay(10);
77 lcd_send_cmd(0x20); // 4bit mode
78 HAL_Delay(10);
79
80 // display initialisation
81 lcd_send_cmd(0x28); // Function set --> DL=0 (4 bit mode), N = 1 (2 line display) F = 0 (5x8 characters)
82 HAL_Delay(10);
83 lcd_send_cmd(0x08); //Display on/off control --> D=0,C=0, B=0 ---> display off
84 HAL_Delay(10);
85 lcd_send_cmd(0x01); // clear display
86 HAL_Delay(20);
87 lcd_send_cmd(0x06); //Entry mode set --> I/D = 1 (increment cursor) & S = 0 (no shift)
88 HAL_Delay(10);
89 lcd_send_cmd(0x0C); //Display on/off control --> D = 1, C and B = 0. (Cursor and blink, last two bits)
90 HAL_Delay(50);
91 lcd_send_cmd(0x80 | 0x43);
92 lcd_send_string("KYC AGRO V1.0");
93 HAL_Delay(10);
94 lcd_create_custom(heart, heartIndex);

112void lcd_send_string(const char *str)
113{
114 uint8_t rep = 195;
115 char *auml = "\xE1";
116 char *ouml = "\xEF";
117 char *uuml = "\xF5";
118 char *sz = "\xE2";
119 bool is_cmd = false;
120
121 while (*str)
122 if (*str != '\0')
123 {
124 if (is_cmd)
125 {
126 switch (*str)
127 {
128 case REP_AUML:
129 lcd_send_data(*auml);
130 break;
131 case REP_OUML:
132 lcd_send_data(*ouml);
133 break;
134 case REP_UUML:
135 lcd_send_data(*uuml);
136 break;
137 case REP_SZ:
138 lcd_send_data(*sz);
139 break;
140 default:
141 break;
142 }
143 *str++;
144 is_cmd = false;
145 continue;
146 }
147 if (*str == rep)
148 {
149 *str++;
150 is_cmd = true;
151 continue;
152 }
153 lcd_send_data(*str++);
154 }
155}
