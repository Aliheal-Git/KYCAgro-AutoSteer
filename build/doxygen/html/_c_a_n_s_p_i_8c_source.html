42/* Entering Sleep Mode */
43void CANSPI_Sleep(void)
44{
45 /* Clear CAN bus wakeup interrupt */
46 MCP2515_BitModify(MCP2515_CANINTF, 0x40, 0x00);
47
48 /* Enable CAN bus activity wakeup */
49 MCP2515_BitModify(MCP2515_CANINTE, 0x40, 0x41);
50
51 MCP2515_SetSleepMode();
52}

109 if(!MCP2515_Initialize())

115
116 /* Change mode as configuration mode */
117 if(!MCP2515_SetConfigMode())

120 }
121
122 /* Configure filter & mask */
123 MCP2515_WriteByteSequence(MCP2515_RXM0SIDH, MCP2515_RXM0EID0, &(RXM0reg.RXM0SIDH));
124 MCP2515_WriteByteSequence(MCP2515_RXM1SIDH, MCP2515_RXM1EID0, &(RXM1reg.RXM1SIDH));

129 MCP2515_WriteByteSequence(MCP2515_RXF4SIDH, MCP2515_RXF4EID0, &(RXF4reg.RXF4SIDH));

132 /* Accept All (Standard + Extended) */
133 MCP2515_WriteByte(MCP2515_RXB0CTRL, 0x04); //Enable BUKT, Accept Filter 0
134 MCP2515_WriteByte(MCP2515_RXB1CTRL, 0x01); //Accept Filter 1
135
136 /*
137 * tq = 2 * (brp(0) + 1) / 16000000 = 0.125us
138 * tbit = (SYNC_SEG(1 fixed) + PROP_SEG + PS1 + PS2)
139 * tbit = 1tq + 5tq + 6tq + 4tq = 16tq
140 * 16tq = 2us = 500kbps

142
143 /* 00(SJW 1tq) 000000 */
144 // 16 Mhz
145 MCP2515_WriteByte(MCP2515_CNF1, 0x00);
146
147 /* 1 1 100(5tq) 101(6tq) */
148 MCP2515_WriteByte(MCP2515_CNF2, 0xF0);
149
150 /* 1 0 000 011(4tq) */
151 MCP2515_WriteByte(MCP2515_CNF3, 0x86);
152 // 8Mhz

159// MCP2515_WriteByte(MCP2515_CNF3, 0x82);

161 /* Normal 모드로 설정 */
162 if(!MCP2515_SetNormalMode())
163 return false;

168/* Transmit CAN message */
169uint8_t CANSPI_Transmit(uCAN_MSG *tempCanMsg)
170{
171 uint8_t returnValue = 0;
172
173 idReg.tempSIDH = 0;
174 idReg.tempSIDL = 0;
175 idReg.tempEID8 = 0;
176 idReg.tempEID0 = 0;
177
178 ctrlStatus.ctrl_status = MCP2515_ReadStatus();
179
180 /* Finding empty buffer */
181 if (ctrlStatus.TXB0REQ != 1)
182 {
183 /* convert CAN ID for register */
184 convertCANid2Reg(tempCanMsg->frame.id, tempCanMsg->frame.idType, &idReg);
185
186 /* Load data to Tx Buffer */
187 MCP2515_LoadTxSequence(MCP2515_LOAD_TXB0SIDH, &(idReg.tempSIDH), tempCanMsg->frame.dlc, &(tempCanMsg->frame.data0));
188
189 /* Request to transmit */
190 MCP2515_RequestToSend(MCP2515_RTS_TX0);
191
192 returnValue = 1;
193 }
194 else if (ctrlStatus.TXB1REQ != 1)
195 {
196 convertCANid2Reg(tempCanMsg->frame.id, tempCanMsg->frame.idType, &idReg);
197
198 MCP2515_LoadTxSequence(MCP2515_LOAD_TXB1SIDH, &(idReg.tempSIDH), tempCanMsg->frame.dlc, &(tempCanMsg->frame.data0));
199 MCP2515_RequestToSend(MCP2515_RTS_TX1);
200
201 returnValue = 1;
202 }
203 else if (ctrlStatus.TXB2REQ != 1)
204 {
205 convertCANid2Reg(tempCanMsg->frame.id, tempCanMsg->frame.idType, &idReg);
206
207 MCP2515_LoadTxSequence(MCP2515_LOAD_TXB2SIDH, &(idReg.tempSIDH), tempCanMsg->frame.dlc, &(tempCanMsg->frame.data0));
208 MCP2515_RequestToSend(MCP2515_RTS_TX2);
209
210 returnValue = 1;
211 }
212
213 return (returnValue);
214}

216/* Receive CAN message */
217uint8_t CANSPI_Receive(uCAN_MSG *tempCanMsg)
218{
219 uint8_t returnValue = 0;
220 rx_reg_t rxReg;
221 ctrl_rx_status_t rxStatus;
222
223 rxStatus.ctrl_rx_status = MCP2515_GetRxStatus();
224
225 /* Check receive buffer */
226 if (rxStatus.rxBuffer != 0)
227 {
228 /* finding buffer which has a message */
229 if ((rxStatus.rxBuffer == MSG_IN_RXB0)|(rxStatus.rxBuffer == MSG_IN_BOTH_BUFFERS))
230 {
231 MCP2515_ReadRxSequence(MCP2515_READ_RXB0SIDH, rxReg.rx_reg_array, sizeof(rxReg.rx_reg_array));
232 }
233 else if (rxStatus.rxBuffer == MSG_IN_RXB1)
234 {
235 MCP2515_ReadRxSequence(MCP2515_READ_RXB1SIDH, rxReg.rx_reg_array, sizeof(rxReg.rx_reg_array));
236 }
237
238 /* if the message is extended CAN type */
239 if (rxStatus.msgType == dEXTENDED_CAN_MSG_ID_2_0B)
240 {
241 tempCanMsg->frame.idType = (uint8_t) dEXTENDED_CAN_MSG_ID_2_0B;
242 tempCanMsg->frame.id = convertReg2ExtendedCANid(rxReg.RXBnEID8, rxReg.RXBnEID0, rxReg.RXBnSIDH, rxReg.RXBnSIDL);
243 }
244 else
245 {
246 /* Standard type */
247 tempCanMsg->frame.idType = (uint8_t) dSTANDARD_CAN_MSG_ID_2_0B;
248 tempCanMsg->frame.id = convertReg2StandardCANid(rxReg.RXBnSIDH, rxReg.RXBnSIDL);
249 }
250
251 tempCanMsg->frame.dlc = rxReg.RXBnDLC;
252 tempCanMsg->frame.data0 = rxReg.RXBnD0;
253 tempCanMsg->frame.data1 = rxReg.RXBnD1;
254 tempCanMsg->frame.data2 = rxReg.RXBnD2;
255 tempCanMsg->frame.data3 = rxReg.RXBnD3;
256 tempCanMsg->frame.data4 = rxReg.RXBnD4;
257 tempCanMsg->frame.data5 = rxReg.RXBnD5;
258 tempCanMsg->frame.data6 = rxReg.RXBnD6;
259 tempCanMsg->frame.data7 = rxReg.RXBnD7;
260
261 returnValue = 1;
262 }
263
264 return (returnValue);
265}

332/* convert register value to extended CAN ID */
333static uint32_t convertReg2ExtendedCANid(uint8_t tempRXBn_EIDH, uint8_t tempRXBn_EIDL, uint8_t tempRXBn_SIDH, uint8_t tempRXBn_SIDL)
334{
335 uint32_t returnValue = 0;
336 uint32_t ConvertedID = 0;
337 uint8_t CAN_standardLo_ID_lo2bits;
338 uint8_t CAN_standardLo_ID_hi3bits;
339
340 CAN_standardLo_ID_lo2bits = (tempRXBn_SIDL & 0x03);
341 CAN_standardLo_ID_hi3bits = (tempRXBn_SIDL >> 5);
342 ConvertedID = (tempRXBn_SIDH << 3);
343 ConvertedID = ConvertedID + CAN_standardLo_ID_hi3bits;
344 ConvertedID = (ConvertedID << 2);
345 ConvertedID = ConvertedID + CAN_standardLo_ID_lo2bits;
346 ConvertedID = (ConvertedID << 8);
347 ConvertedID = ConvertedID + tempRXBn_EIDH;
348 ConvertedID = (ConvertedID << 8);
349 ConvertedID = ConvertedID + tempRXBn_EIDL;
350 returnValue = ConvertedID;
351 return (returnValue);
352}

354/* convert register value to standard CAN ID */
355static uint32_t convertReg2StandardCANid(uint8_t tempRXBn_SIDH, uint8_t tempRXBn_SIDL)
356{
357 uint32_t returnValue = 0;
358 uint32_t ConvertedID;
359
360 ConvertedID = (tempRXBn_SIDH << 3);
361 ConvertedID = ConvertedID + (tempRXBn_SIDL >> 5);
362 returnValue = ConvertedID;
363
364 return (returnValue);
365}

367/* convert CAN ID to register value */
368static void convertCANid2Reg(uint32_t tempPassedInID, uint8_t canIdType, id_reg_t *passedIdReg)
369{
370 uint8_t wipSIDL = 0;
371
372 if (canIdType == dEXTENDED_CAN_MSG_ID_2_0B)
373 {
374 //EID0
375 passedIdReg->tempEID0 = 0xFF & tempPassedInID;
376 tempPassedInID = tempPassedInID >> 8;
377
378 //EID8
379 passedIdReg->tempEID8 = 0xFF & tempPassedInID;
380 tempPassedInID = tempPassedInID >> 8;
381
382 //SIDL
383 wipSIDL = 0x03 & tempPassedInID;
384 tempPassedInID = tempPassedInID << 3;
385 wipSIDL = (0xE0 & tempPassedInID) + wipSIDL;
386 wipSIDL = wipSIDL + 0x08;
387 passedIdReg->tempSIDL = 0xEB & wipSIDL;
388
389 //SIDH
390 tempPassedInID = tempPassedInID >> 8;
391 passedIdReg->tempSIDH = 0xFF & tempPassedInID;
392 }
393 else
394 {
395 passedIdReg->tempEID8 = 0;
396 passedIdReg->tempEID0 = 0;
397 tempPassedInID = tempPassedInID << 5;
398 passedIdReg->tempSIDL = 0xFF & tempPassedInID;
399 tempPassedInID = tempPassedInID >> 8;
400 passedIdReg->tempSIDH = 0xFF & tempPassedInID;
401 }
402}
