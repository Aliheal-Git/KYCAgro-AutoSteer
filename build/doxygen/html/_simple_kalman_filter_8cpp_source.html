17float SimpleKalmanFilter::updateEstimate(float mea)
18{
19 _kalman_gain = _err_estimate / (_err_estimate + _err_measure);
20 _current_estimate = _last_estimate + _kalman_gain * (mea - _last_estimate);
21 _err_estimate = (1.0f - _kalman_gain) * _err_estimate + fabsf(_last_estimate - _current_estimate) * _q;
22 _last_estimate = _current_estimate;
23
24 return _current_estimate;
25}
