11 *
12 * The gain values act as multipliers for their corresponding components of PID
13 * (more detail later). The target is the value which the system strives to
14 * reach by manipulating the output. The feedback is the system's actual
15 * position or status in regards to the physical world.
16 * Another important term in PID is "error", which refers to the difference
17 * between the target and the feedback.
18 *
19 * Each of the three components of PID contributes a unique behavior to the
20 * system.
21 *
22 * The Proportional component introduces a linear relationship between the
23 * error (target minus feedback) and the output. This means that as the
24 * feedback grows further away from the target, the output grows
25 * proportionally stronger.
26 *
27 * Proportional component = (P Gain) * (target - feedback)
28 *
29 * The Integral component is designed to give a very precise approach of the
30 * feedback to the target. Depending on the scale of the physical system
31 * and the precision of feedback (e.g. sensors), the proportional component
32 * alone is likely not sufficient to provide adequate power (e.g. to motors)
33 * to guide the system in regards to small-scale corrections. The Integral
34 * component integrates the error of the system (target - feedback) over
35 * time. If the system reaches a point where it is close but not exactly
36 * on top of the target, the integration will slowly build until it is
37 * powerful enough to overcome static resistances and move the system
38 * preciesly to the target.
39 *
40 * Integral component = (I Gain) * Integral of error over time
41 *
42 * *In this implementation, Integral is calculated with a running
43 * summation of the system's error, updated at each tick.
44 *
45 * The Derivative component measures the rate of change of the feedback.
46 * It can reduce the strength of the output if the feedback is approaching
47 * the target too quickly or if the feedback is moving away from the target.
48 *
49 * Derivative component = (D Gain) * ((error - lastError) / time - lastTime)
50 *
51 * The output generated by the PID Controller is the sum of the three
52 * components.
53 *
54 * PID output = Proportional component + Integral component + Derivative component
55 */

134template <class T>
135T PIDController<T>::tick()
136{
137 currentFeedback = _pidSource();
138 if(enabled)
139 {
140 //Retrieve system feedback from user callback.
141
142
143 //Apply input bounds if necessary.
144 if(inputBounded)
145 {
146 if(currentFeedback > inputUpperBound) currentFeedback = inputUpperBound;
147 if(currentFeedback < inputLowerBound) currentFeedback = inputLowerBound;
148 }
149
150 /*
151 * Feedback wrapping causes two distant numbers to appear adjacent to one
152 * another for the purpose of calculating the system's error.
153 */
154 if(feedbackWrapped)
155 {
156 /*
157 * There are three ways to traverse from one point to another in this setup.
158 *
159 * 1) Target --> Feedback
160 *
161 * The other two ways involve bridging a gap connected by the upper and
162 * lower bounds of the feedback wrap.
163 *
164 * 2) Target --> Upper Bound == Lower Bound --> Feedback
165 *
166 * 3) Target --> Lower Bound == Upper Bound --> Feedback
167 *
168 * Of these three paths, one should always be shorter than the other two,
169 * unless all three are equal, in which case it does not matter which path
170 * is taken.
171 */
172 float regErr = target - currentFeedback;
173 float altErr1 = (target - feedbackWrapLowerBound) + (feedbackWrapUpperBound - currentFeedback);
174 float altErr2 = (feedbackWrapUpperBound - target) + (currentFeedback - feedbackWrapLowerBound);
175
176 //Calculate the absolute values of each error.
177 float regErrAbs = (regErr >= 0) ? regErr : -regErr;
178 float altErr1Abs = (altErr1 >= 0) ? altErr1 : -altErr1;
179 float altErr2Abs = (altErr2 >= 0) ? altErr2 : -altErr2;
180
181 //Use the error with the smallest absolute value
182 if(regErrAbs <= altErr1Abs && regErrAbs <= altErr2Abs) //If reguErrAbs is smallest
183 {
184 error = regErr;
185 }
186 else if(altErr1Abs < regErrAbs && altErr1Abs < altErr2Abs) //If altErr1Abs is smallest
187 {
188 error = altErr1Abs;
189 }
190 else if(altErr2Abs < regErrAbs && altErr2Abs < altErr1Abs) //If altErr2Abs is smallest
191 {
192 error = altErr2Abs;
193 }
194 }
195 else
196 {
197 //Calculate the error between the feedback and the target.
198 error = target - currentFeedback;
199 }
200
201 //If we have a registered way to retrieve the system time, use time in PID calculations.
202 if(timeFunctionRegistered)
203 {
204 //Retrieve system time
205 currentTime = _getSystemTime();
206 if (currentTime == lastTime)
207 error = 0;
208
209 //Calculate time since last tick() cycle.
210 long deltaTime = currentTime - lastTime;
211
212 //Calculate the integral of the feedback data since last cycle.
213 int cycleIntegral = ((lastError + error) / 2) * deltaTime;
214
215 //Add this cycle's integral to the integral cumulation.
216 integralCumulation += cycleIntegral;
217 if (fabs(error) < 0.08)
218 integralCumulation = 0;
219
220 //Calculate the slope of the line with data from the current and last cycles.
221 cycleDerivative = (error - lastError) / deltaTime;
222
223 //Save time data for next iteration.
224 lastTime = currentTime;
225 }
226 //If we have no way to retrieve system time, estimate calculations.
227 else
228 {
229 integralCumulation += error;
230 cycleDerivative = (error - lastError);
231 }
232
233 //Prevent the integral cumulation from becoming overwhelmingly huge.
234 if(integralCumulation > maxCumulation) integralCumulation = maxCumulation;
235 if(integralCumulation < -maxCumulation) integralCumulation = -maxCumulation;
236
237 //Calculate the system output based on data and PID gains.
238 _peffect = (error * _p);
239 _ieffect = (integralCumulation * _i);
240 _deffect = (cycleDerivative * _d);
241 output = (float) ((error * _p) + (integralCumulation * _i) + (cycleDerivative * _d));
242
243 //Save a record of this iteration's data.
244 lastFeedback = currentFeedback;
245 lastError = error;
246
247 //Trim the output to the bounds if needed.
248 if(outputBounded)
249 {
250 if(output > outputUpperBound) output = outputUpperBound;
251 if(output < outputLowerBound) output = outputLowerBound;
252 }
253
254 _pidOutput(output);
255 }
256 return currentFeedback;
257}
